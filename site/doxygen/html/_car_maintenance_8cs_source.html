103 public int FileAppend(string FileName, string text)
104 {
105 try
106 {
107 string lastLine = "";
108 string currentLine = "";
109 char character;
110
111 using (myFile = new FileStream(FileName, FileMode.Open, FileAccess.Read))
112 {
113 using (StreamReader streamReader = new StreamReader(myFile, Encoding.Default))
114 {
115 while (!streamReader.EndOfStream)
116 {
117 character = (char)streamReader.Read();
118 if (character == '\n')
119 {
120 currentLine += character;
121 lastLine = currentLine;
122 currentLine = "";
123 continue;
124 }
125
126 currentLine += character;
127 }
128 }
129 }
130

133 text = lineNumber + "-)" + text + "\n";
134
135 using (myFile = new FileStream(FileName, FileMode.Append, FileAccess.Write))
136 {
137 using (StreamWriter streamWriter = new StreamWriter(myFile, Encoding.Default))
138 {
139 streamWriter.Write(text);

141 }

144 }

292 streamWriter.Write(updatedLineWithNewNumber);
